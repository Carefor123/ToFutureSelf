ToFutureSelf 2020-07-17 17:30:55 循环遍历 JavaScript

# 1. for循环 --- 使用率最高，最基本的一种遍历方式

var arr = [
  {id:11, name:'龙爪堰'},
  {id:22, name:'金楠天街'},
  {id:33, name:'外三环'}
];
var arrIds = [];
for(var i=0,len=arr.length;i<len; i++){
    arrIds.push(arr[i].id);
}
console.log(arrIds); // [11,22,33]

# 2. forEach循环 --- 传入要执行的回调函数，函数有三个参数；第一个参数为数组元素(必选)，第二个参数为数组元素索引值(可选)，第三个参数为数组本身(可选)

var arr = [
  {id:11, name:'龙爪堰'},
  {id:22, name:'金楠天街'},
  {id:33, name:'外三环'}
];
var arrIds = [];
arr.forEach((item,key,arr) => {
  arrIds.push(item.id);
});
console.log(arrIds); // [11,22,33]

# 3. for in循环 --- 循环可用于循环对象和数组,推荐用于循环对象,可以用来遍历JSON

var arr = [
  {id:11, name:'龙爪堰'},
  {id:22, name:'金楠天街'},
  {id:33, name:'外三环'}
];
var arrIds = [];
for(var key in arr){
  arrIds.push(arr[key].id);
};
console.log(arrIds); // [11,22,33]

# 4. for of循环 --- 可循环数组和对象，推荐用于遍历数组。for...of提供了三个新方法：key()是对键名的遍历；value()是对键值的遍历；entries()是对键值对的遍历

var arr=['龙爪堰','金楠天街','外三环'];
for (var item of arr) {  
  console.log(item); // 龙爪堰, 金楠天街, 外三环
}
// 输出数组索引
for (var item of arr.keys()) {  
  console.log(item);  // 0, 1, 2
}
// 输出内容和索引
for (var [item, val] of arr.entries()) {  
  console.log(item + ':' + val); // 0:龙爪堰; 1:金楠天街; 2:外三环
}
